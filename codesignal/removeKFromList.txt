// Singly-linked lists are already defined with this interface:

// }
// T value;
// ListNode<T> next;
// }

ListNode<Integer> removeKFromList(ListNode<Integer> l, int k) {
    ListNode<Integer> listaTeste = new ListNode<>(null);
    listaTeste.next = l;

    ListNode<Integer> atual = l;
    ListNode<Integer> proximo = listaTeste;
    while (atual != null) {
        if (atual.value == k) {
            proximo.next = atual.next;
            atual = atual.next;
        } else {
            proximo = atual;
            atual = atual.next;
        }
    }
    return listaTeste.next;
}
